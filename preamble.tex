% currently this works as \begin{env}[optional prop/def]{x.y.z}
\makeatletter
\newenvironment{env}[2][\@nil]{%
    \def\tmp{#1}%
    \ifx\tmp\@nnil
        \par\medskip\noindent\indent\textbf{(#2)}\rmfamily
    \else
        \par\medskip\noindent\indent\textit{#1}~\textbf{(#2)}.\,---\itshape
    \fi}
\makeatother

}
\makeatother
